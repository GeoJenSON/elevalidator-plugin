280    /**
281     * Fix selected errors
282     */
283    private void fixErrors() {
284        TreePath[] selectionPaths = tree.getSelectionPaths();
285        if (selectionPaths == null)
286            return;
287
288        Set<DefaultMutableTreeNode> processedNodes = new HashSet<>();
289
290        List<TestError> errorsToFix = new LinkedList<>();
291        for (TreePath path : selectionPaths) {
292            DefaultMutableTreeNode node = (DefaultMutableTreeNode) path.getLastPathComponent();
293            if (node != null) {
294                ValidatorTreePanel.visitTestErrors(node, errorsToFix::add, processedNodes);
295            }
296        }
297
298        // run fix task asynchronously
299        MainApplication.worker.submit(new FixTask(errorsToFix));
300    }

302    /**
303     * Set selected errors to ignore state
304     */
305    private void ignoreErrors() {
306        int asked = JOptionPane.DEFAULT_OPTION;
307        AtomicBoolean changed = new AtomicBoolean();
308        TreePath[] selectionPaths = tree.getSelectionPaths();
309        if (selectionPaths == null)
310            return;
311
312        Set<DefaultMutableTreeNode> processedNodes = new HashSet<>();
313        for (TreePath path : selectionPaths) {
314            DefaultMutableTreeNode node = (DefaultMutableTreeNode) path.getLastPathComponent();
315            if (node == null) {
316                continue;
317            }
318
319            Object mainNodeInfo = node.getUserObject();
320            final int depth = node.getDepth();
321            if (!(mainNodeInfo instanceof TestError)) {
322                Set<Pair<String, String>> state = new HashSet<>();
323                // ask if the whole set should be ignored
324                if (asked == JOptionPane.DEFAULT_OPTION) {
325                    String[] a = {tr("Whole group"), tr("Single elements"), tr("Nothing")};
326                    asked = JOptionPane.showOptionDialog(MainApplication.getMainFrame(), tr("Ignore whole group or individual elements?"),
327                            tr("Ignoring elements"), JOptionPane.YES_NO_CANCEL_OPTION, JOptionPane.WARNING_MESSAGE, null,
328                            a, a[1]);
329                }
330                if (asked == JOptionPane.YES_NO_OPTION) {
331                    ValidatorTreePanel.visitTestErrors(node, err -> {
332                        err.setIgnored(true);
333                        changed.set(true);
334                        state.add(new Pair<>(depth == 1 ? err.getIgnoreSubGroup() : err.getIgnoreGroup(), err.getMessage()));
335                    }, processedNodes);
336                    for (Pair<String, String> s : state) {
337                        OsmValidator.addIgnoredError(s.a, s.b);
338                    }
339                    continue;
340                } else if (asked == JOptionPane.CANCEL_OPTION || asked == JOptionPane.CLOSED_OPTION) {
341                    continue;
342                }
343            }
344
345            ValidatorTreePanel.visitTestErrors(node, error -> {
346                String state = error.getIgnoreState();
347                if (state != null) {
348                    OsmValidator.addIgnoredError(state, error.getMessage());
349                }
350                changed.set(true);
351                error.setIgnored(true);
352            }, processedNodes);
353        }
354        if (changed.get()) {
355            tree.resetErrors();
356            OsmValidator.saveIgnoredErrors();
357            invalidateValidatorLayers();
358        }
359    }

361    /**
362     * Sets the selection of the map to the current selected items.
363     */
364    private void setSelectedItems() {
365        DataSet ds = MainApplication.getLayerManager().getActiveDataSet();
366        if (tree == null || ds == null)
367            return;
368
369        TreePath[] selectedPaths = tree.getSelectionPaths();
370        if (selectedPaths == null)
371            return;
372
373        Collection<OsmPrimitive> sel = new HashSet<>(40);
374        for (TreePath path : selectedPaths) {
375            DefaultMutableTreeNode node = (DefaultMutableTreeNode) path.getLastPathComponent();
376            Enumeration<?> children = node.breadthFirstEnumeration();
377            while (children.hasMoreElements()) {
378                DefaultMutableTreeNode childNode = (DefaultMutableTreeNode) children.nextElement();
379                Object nodeInfo = childNode.getUserObject();
380                if (nodeInfo instanceof TestError) {
381                    TestError error = (TestError) nodeInfo;
382                    error.getPrimitives().stream()
383                            .filter(OsmPrimitive::isSelectable)
384                            .forEach(sel::add);
385                }
386            }
387        }
388        ds.setSelected(sel);
389    }

391    /**
392     * Checks for fixes in selected element and, if needed, adds to the sel
393     * parameter all selected elements
394     *
395     * @param sel
396     *            The collection where to add all selected elements
397     * @param addSelected
398     *            if true, add all selected elements to collection
399     * @return whether the selected elements has any fix
400     */
401    private boolean setSelection(Collection<OsmPrimitive> sel, boolean addSelected) {
402        AtomicBoolean hasFixes = new AtomicBoolean();
403
404        DefaultMutableTreeNode node = (DefaultMutableTreeNode) tree.getLastSelectedPathComponent();
405        if (lastSelectedNode != null && !lastSelectedNode.equals(node)) {
406            ValidatorTreePanel.visitTestErrors(lastSelectedNode, error -> error.setSelected(false));
407        }
408
409        lastSelectedNode = node;
410        if (node != null) {
411            ValidatorTreePanel.visitTestErrors(node, error -> {
412                error.setSelected(true);
413
414                hasFixes.set(hasFixes.get() || error.isFixable());
415                if (addSelected) {
416                    error.getPrimitives().stream()
417                            .filter(OsmPrimitive::isSelectable)
418                            .forEach(sel::add);
419                }
420            });
421            selectAction.setEnabled(true);
422            if (ignoreAction != null) {
423                ignoreAction.setEnabled(!(node.getUserObject() instanceof Severity));
424            }
425        }
426
427        return hasFixes.get();
428    }

483    /**
484     * Watches for double clicks and launches the popup menu.
485     */
486    class MouseEventHandler extends PopupMenuLauncher {
487
488        MouseEventHandler() {
489            super(popupMenu);
490        }
491
492        @Override
493        public void mouseClicked(MouseEvent e) {
494            TreePath selPath = tree.getPathForLocation(e.getX(), e.getY());
495            if (selPath == null) {
496                tree.clearSelection();
497            }
498
499            fixAction.setEnabled(false);
500            if (ignoreAction != null) {
501                ignoreAction.setEnabled(false);
502            }
503            selectAction.setEnabled(false);
504
505            boolean isDblClick = isDoubleClick(e);
506
507            Collection<OsmPrimitive> sel = isDblClick ? new HashSet<>(40) : null;
508
509            boolean hasFixes = setSelection(sel, isDblClick);
510            fixAction.setEnabled(hasFixes);
511
512            if (isDblClick) {
513                DataSet ds = MainApplication.getLayerManager().getActiveDataSet();
514                if (ds != null) {
515                    ds.setSelected(sel);
516                }
517                if (Config.getPref().getBoolean("validator.autozoom", false)) {
518                    AutoScaleAction.zoomTo(sel);
519                }
520            }
521        }
522
523        @Override
524        public void launch(MouseEvent e) {
525            TreePath selPath = tree.getPathForLocation(e.getX(), e.getY());
526            if (selPath == null)
527                return;
528            DefaultMutableTreeNode node = (DefaultMutableTreeNode) selPath.getPathComponent(selPath.getPathCount() - 1);
529            if (!(node.getUserObject() instanceof TestError))
530                return;
531            super.launch(e);
532        }
533    }

554    /**
555     * A visitor that is used to compute the bounds of an error.
556     */
557    public static class ValidatorBoundingXYVisitor extends BoundingXYVisitor implements ValidatorVisitor {
558        @Override
559        public void visit(OsmPrimitive p) {
560            if (p.isUsable()) {
561                p.accept((PrimitiveVisitor) this);
562            }
563        }
564
565        @Override
566        public void visit(WaySegment ws) {
567            if (ws.lowerIndex < 0 || ws.lowerIndex + 1 >= ws.way.getNodesCount())
568                return;
569            visit(ws.getFirstNode());
570            visit(ws.getSecondNode());
571        }
572
573        @Override
574        public void visit(List<Node> nodes) {
575            for (Node n: nodes) {
576                visit(n);
577            }
578        }
579
580        @Override
581        public void visit(TestError error) {
582            if (error != null) {
583                error.visitHighlighted(this);
584            }
585        }
586    }

607    /**
608     * Task for fixing a collection of {@link TestError}s. Can be run asynchronously.
609     */
610    class FixTask extends PleaseWaitRunnable {
611        private final Collection<TestError> testErrors;
612        private final List<Command> fixCommands = new ArrayList<>();
613        private boolean canceled;
614
615        FixTask(Collection<TestError> testErrors) {
616            super(tr("Fixing errors ..."), false /* don't ignore exceptions */);
617            this.testErrors = testErrors == null ? new ArrayList<>() : testErrors;
618        }
619
620        @Override
621        protected void cancel() {
622            this.canceled = true;
623        }
624
625        @Override
626        protected void finish() {
627            // do nothing
628        }
629
630        protected void fixError(TestError error) throws InterruptedException, InvocationTargetException {
631            if (error.isFixable()) {
632                if (error.getPrimitives().stream().noneMatch(p -> p.isDeleted() || p.getDataSet() == null)) {
633                    final Command fixCommand = error.getFix();
634                    if (fixCommand != null) {
635                        SwingUtilities.invokeAndWait(fixCommand::executeCommand);
636                        fixCommands.add(fixCommand);
637                    }
638                }
639                // It is wanted to ignore an error if it said fixable, even if fixCommand was null
640                // This is to fix #5764 and #5773:
641                // a delete command, for example, may be null if all concerned primitives have already been deleted
642                error.setIgnored(true);
643            }
644        }
645
646        @Override
647        protected void realRun() throws SAXException, IOException, OsmTransferException {
648            ProgressMonitor monitor = getProgressMonitor();
649            try {
650                monitor.setTicksCount(testErrors.size());
651                final DataSet ds = MainApplication.getLayerManager().getActiveDataSet();
652                int i = 0;
653                SwingUtilities.invokeAndWait(ds::beginUpdate);
654                tree.setResetScheduled();
655                try {
656                    for (TestError error: testErrors) {
657                        i++;
658                        monitor.subTask(tr("Fixing ({0}/{1}): ''{2}''", i, testErrors.size(), error.getMessage()));
659                        if (this.canceled)
660                            return;
661                        fixError(error);
662                        monitor.worked(1);
663                    }
664                } finally {
665                    SwingUtilities.invokeAndWait(ds::endUpdate);
666                }
667                monitor.subTask(tr("Updating map ..."));
668                SwingUtilities.invokeAndWait(() -> {
669                    if (!fixCommands.isEmpty()) {
670                        UndoRedoHandler.getInstance().add(
671                                fixCommands.size() > 1 ? new AutofixCommand(fixCommands) : fixCommands.get(0), false);
672                    }
673                    invalidateValidatorLayers();
674                });
675            } catch (InterruptedException e) {
676                tryUndo();
677                throw new JosmRuntimeException(e);
678            } catch (InvocationTargetException e) {
679                // FIXME: signature of realRun should have a generic checked exception we could throw here
680                throw new JosmRuntimeException(e);
681            } finally {
682                if (monitor.isCanceled()) {
683                    tryUndo();
684                }
685                GuiHelper.runInEDTAndWait(tree::resetErrors);
686                monitor.finishTask();
687            }
688        }
689
690        /**
691         * Undo commands as they were not yet added to the UndoRedo Handler
692         */
693        private void tryUndo() {
694            MainApplication.getLayerManager().getActiveDataSet().update(() -> {
695                for (int i = fixCommands.size() - 1; i >= 0; i--) {
696                    fixCommands.get(i).undoCommand();
697                }
698            });
699        }
