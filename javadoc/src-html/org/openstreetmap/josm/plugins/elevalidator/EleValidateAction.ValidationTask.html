035/**
036 * The action that does the validate thing.
037 * <p>
038 * This action iterates through all active tests and give them the data, so that
039 * each one can test it.
040 *
041 * @author frsantos
042 */
043public class EleValidateAction extends JosmAction {
044
045    /** Last selection used to validate */
046    private transient Collection<OsmPrimitive> lastSelection;
047
048    /**
049     * Constructor
050     */
051    public EleValidateAction() {
052        super(tr("EleValidation"), "dialogs/validator", tr("Performs the elevator validation"),
053                Shortcut.registerShortcut("tools:validate", tr("Tool: {0}", tr("Validation")),
054                        KeyEvent.VK_V, Shortcut.SHIFT), true);
055    }
056
057    @Override
058    public void actionPerformed(ActionEvent ev) {
059        doValidate(true);
060    }
061
062    /**
063     * Does the validation.
064     * <p>
065     * If getSelectedItems is true, the selected items (or all items, if no one
066     * is selected) are validated. If it is false, last selected items are revalidated
067     *
068     * @param getSelectedItems If selected or last selected items must be validated
069     */
070    public void doValidate(boolean getSelectedItems) {
071        MapFrame map = MainApplication.getMap();
072        if (map == null || !map.isVisible())
073            return;
074
075        ElevatorValidator.initializeTests();
076
077        Collection<Test> tests = ElevatorValidator.getEnabledTests(false);
078        if (tests.isEmpty())
079            return;
080
081        Collection<OsmPrimitive> selection;
082        if (getSelectedItems) {
083            selection = getLayerManager().getActiveDataSet().getAllSelected();
084            if (selection.isEmpty()) {
085                selection = getLayerManager().getActiveDataSet().allNonDeletedPrimitives();
086                lastSelection = null;
087            } else {
088                AggregatePrimitivesVisitor v = new AggregatePrimitivesVisitor();
089                selection = v.visit(selection);
090                lastSelection = selection;
091            }
092        } else {
093            selection = Optional.ofNullable(lastSelection).orElseGet(
094                    () -> getLayerManager().getActiveDataSet().allNonDeletedPrimitives());
095        }
096
097        MainApplication.worker.submit(new ValidationTask(tests, selection, lastSelection));
098    }
099
100    @Override
101    public void updateEnabledState() {
102        setEnabled(getLayerManager().getActiveDataSet() != null);
103    }
104
105    @Override
106    public void destroy() {
107        // Hack - this action should stay forever because it could be added to toolbar
108        // Do not call super.destroy() here
109        lastSelection = null;
110    }
111
112    /**
113     * Asynchronous task for running a collection of tests against a collection of primitives
114     */
115    static class ValidationTask extends PleaseWaitRunnable {
116        private Collection<Test> tests;
117        private final Collection<OsmPrimitive> validatedPrimitives;
118        private final Collection<OsmPrimitive> formerValidatedPrimitives;
119        private boolean canceled;
120        private List<TestError> errors;
121
122        /**
123         * Constructs a new {@code ValidationTask}
124         * @param tests  the tests to run
125         * @param validatedPrimitives the collection of primitives to validate.
126         * @param formerValidatedPrimitives the last collection of primitives being validates. May be null.
127         */
128        ValidationTask(Collection<Test> tests, Collection<OsmPrimitive> validatedPrimitives,
129                       Collection<OsmPrimitive> formerValidatedPrimitives) {
130            super(tr("Validating"), false /*don't ignore exceptions */);
131            this.validatedPrimitives = validatedPrimitives;
132            this.formerValidatedPrimitives = formerValidatedPrimitives;
133            this.tests = tests;
134        }
135
136        @Override
137        protected void cancel() {
138            this.canceled = true;
139        }
140
141        @Override
142        protected void finish() {
143            if (canceled) return;
144
145            // update GUI on Swing EDT
146            //
147            GuiHelper.runInEDT(() -> {
148                ElevalidatorDialog elevalidatorDialog = ElevalidatorPlugin.getElevalidatorDialog();
149                elevalidatorDialog.unfurlDialog();
150                elevalidatorDialog.tree.setErrors(errors);
151                //FIXME: nicer way to find / invalidate the corresponding error layer
152                MainApplication.getLayerManager().getLayersOfType(ValidatorLayer.class).forEach(ValidatorLayer::invalidate);
153                if (!errors.isEmpty()) {
154                    ElevatorValidator.initializeErrorLayer();
155                }
156            });
157        }
158
159        @Override
160        protected void realRun() throws SAXException, IOException,
161                OsmTransferException {
162            if (tests == null || tests.isEmpty())
163                return;
164            errors = new ArrayList<>();
165            getProgressMonitor().setTicksCount(tests.size() * validatedPrimitives.size());
166            int testCounter = 0;
167            for (Test test : tests) {
168                if (canceled)
169                    return;
170                testCounter++;
171                getProgressMonitor().setCustomText(tr("Test {0}/{1}: Starting {2}", testCounter, tests.size(), test.getName()));
172                test.setBeforeUpload(false);
173                test.setPartialSelection(formerValidatedPrimitives != null);
174                test.startTest(getProgressMonitor().createSubTaskMonitor(validatedPrimitives.size(), false));
175                test.visit(validatedPrimitives);
176                test.endTest();
177                errors.addAll(test.getErrors());
178                test.clear();
179            }
180            tests = null;
181            if (Boolean.TRUE.equals(ValidatorPrefHelper.PREF_USE_IGNORE.get())) {
182                getProgressMonitor().setCustomText("");
183                getProgressMonitor().subTask(tr("Updating ignored errors ..."));
184                for (TestError error : errors) {
185                    if (canceled) return;
186                    error.updateIgnored();
187                }
188            }
189        }
190    }
