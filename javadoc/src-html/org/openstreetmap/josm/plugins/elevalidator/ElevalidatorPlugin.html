001package org.openstreetmap.josm.plugins.elevalidator;
002
003import org.openstreetmap.josm.gui.IconToggleButton;
004import org.openstreetmap.josm.gui.MapFrame;
005import org.openstreetmap.josm.plugins.Plugin;
006import org.openstreetmap.josm.plugins.PluginInformation;
007
008public class ElevalidatorPlugin extends Plugin {
009
010    // Will be invoked by JOSM to bootstrap the plugin
011
012    protected static ElevalidatorDialog elevalidatorDialog;
013
014    // @param info  information about the plugin and its local installation */
015    public ElevalidatorPlugin(PluginInformation info) {
016        super(info);
017        // init your plugin
018    }
019
020    @Override
021    public void mapFrameInitialized(MapFrame oldFrame, MapFrame newFrame) {
022        super.mapFrameInitialized(oldFrame, newFrame);
023
024        if (oldFrame == null && newFrame != null) {
025            newFrame.addToggleDialog(elevalidatorDialog = new ElevalidatorDialog());
026        }
027    }
028}
